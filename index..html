// Salva i dati ad ogni modifica document.querySelectorAll('.inp-field').forEach(input => { input.addEventListener('input', () => { localStorage.setItem(input.id, input.value); }); }); // Recupera i dati all'avvio window.addEventListener('load', () => { document.querySelectorAll('.inp-field').forEach(input => { const saved = localStorage.getItem(input.id); if (saved) input.value = saved; }); });
